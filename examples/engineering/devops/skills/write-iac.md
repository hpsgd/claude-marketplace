```hcl
# Resource Graph:
# aws_s3_bucket
#   → aws_s3_bucket_versioning
#   → aws_s3_bucket_server_side_encryption_configuration
#   → aws_s3_bucket_lifecycle_configuration
#   → aws_s3_bucket_public_access_block

locals {
  bucket_name = "${var.environment}-${var.service}-documents"
  common_tags = {
    environment = var.environment
    team        = var.team
    service     = var.service
    managed-by  = "terraform"
  }
}

resource "aws_s3_bucket" "documents" {
  bucket = local.bucket_name
  tags   = local.common_tags
}

resource "aws_s3_bucket_versioning" "documents" {
  bucket = aws_s3_bucket.documents.id
  versioning_configuration {
    status = "Enabled"
  }
}

resource "aws_s3_bucket_server_side_encryption_configuration" "documents" {
  bucket = aws_s3_bucket.documents.id
  rule {
    apply_server_side_encryption_by_default {
      sse_algorithm = "AES256"
    }
  }
}

resource "aws_s3_bucket_lifecycle_configuration" "documents" {
  bucket = aws_s3_bucket.documents.id
  rule {
    id     = "transition-to-infrequent-access"
    status = "Enabled"
    transition {
      days          = var.ia_transition_days
      storage_class = "STANDARD_IA"
    }
    transition {
      days          = var.glacier_transition_days
      storage_class = "GLACIER"
    }
    noncurrent_version_expiration {
      noncurrent_days = var.noncurrent_version_retention_days
    }
  }
}

resource "aws_s3_bucket_public_access_block" "documents" {
  bucket                  = aws_s3_bucket.documents.id
  block_public_acls       = true
  block_public_policy     = true
  ignore_public_acls      = true
  restrict_public_buckets = true
}

resource "aws_s3_bucket_policy" "enforce_tls" {
  bucket = aws_s3_bucket.documents.id
  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Sid       = "DenyNonTLS"
      Effect    = "Deny"
      Principal = "*"
      Action    = "s3:*"
      Resource  = [
        aws_s3_bucket.documents.arn,
        "${aws_s3_bucket.documents.arn}/*"
      ]
      Condition = {
        Bool = { "aws:SecureTransport" = "false" }
      }
    }]
  })
}
```

```hcl
variable "environment" {
  description = "Deployment environment (dev, staging, prod)"
  type        = string
  validation {
    condition     = contains(["dev", "staging", "prod"], var.environment)
    error_message = "Environment must be one of: dev, staging, prod."
  }
}

variable "service" {
  description = "Service name used in resource naming (e.g. 'inventory', 'billing')"
  type        = string
  validation {
    condition     = can(regex("^[a-z][a-z0-9-]{1,20}$", var.service))
    error_message = "Service name must be lowercase alphanumeric with hyphens, 2-21 characters."
  }
}

variable "team" {
  description = "Owning team name for cost allocation and tagging"
  type        = string
}

variable "ia_transition_days" {
  description = "Number of days before transitioning objects to STANDARD_IA storage class"
  type        = number
  default     = 90
  validation {
    condition     = var.ia_transition_days >= 30
    error_message = "S3 minimum transition period to STANDARD_IA is 30 days."
  }
}

variable "glacier_transition_days" {
  description = "Number of days before transitioning objects to GLACIER storage class"
  type        = number
  default     = 365
  validation {
    condition     = var.glacier_transition_days > var.ia_transition_days
    error_message = "Glacier transition must occur after STANDARD_IA transition."
  }
}

variable "noncurrent_version_retention_days" {
  description = "Number of days to retain non-current object versions before expiry"
  type        = number
  default     = 30
}
```

````markdown
# Module: document-storage

Provisions an S3 bucket configured for document storage with versioning, AES-256 server-side encryption, tiered lifecycle transitions (STANDARD_IA → GLACIER), public access blocking, and a TLS-enforcement bucket policy.

## Usage

```hcl
module "documents" {
  source = "./modules/document-storage"

  environment = "prod"
  service     = "inventory"
  team        = "engineering"
}
```

| team | string | — | Owning team for cost tagging |
| ia_transition_days | number | 90 | Days before transition to STANDARD_IA |
| glacier_transition_days | number | 365 | Days before transition to GLACIER |
| noncurrent_version_retention_days | number | 30 | Retention period for non-current versions |

## Outputs

| Name | Description |
|---|---|
| bucket_id | The S3 bucket name |
| bucket_arn | The S3 bucket ARN (use in IAM policies) |

## Prerequisites

- AWS credentials with `s3:*` permissions on the target account
- Terraform state backend bucket `company-terraform-state` must exist
- DynamoDB lock table `terraform-lock` must exist
````
