```csharp
public static class CreateOrderEndpoint
{
    public static async Task<ProblemDetails?> LoadAsync(
        CreateOrderCommand command,
        IDocumentSession session,
        CancellationToken ct)
    {
        var customer = await session.LoadAsync<Customer>(command.CustomerId, ct);
        if (customer is null)
            return new ProblemDetails { Status = 404, Title = "Customer not found" };

        var activeOrderCount = await session.Query<Order>()
            .CountAsync(o => o.CustomerId == command.CustomerId
                          && o.Status == OrderStatus.Active, ct);

        if (activeOrderCount >= 50)
            return new ProblemDetails
            {
                Status = 422,
                Title = "Order limit reached",
                Detail = "A customer cannot have more than 50 active orders."
            };

        foreach (var item in command.LineItems)
        {
            if (item.Quantity is < 1 or > 100)
                return new ProblemDetails
                {
                    Status = 422,
                    Title = "Invalid quantity",
                    Detail = $"Line item quantity must be between 1 and 100. Got: {item.Quantity}"
                };
        }

        return null;
    }

    [WolverinePost("/api/customers/{customerId}/orders")]
    public static OrderPlaced Handle(CreateOrderCommand command)
    {
        var orderId = CombGuidIdGeneration.NewGuid();
        var lineItems = command.LineItems
            .Select(li => new LineItem(li.ProductId, li.Quantity, li.UnitPrice))
            .ToList();

        return new OrderPlaced(orderId, command.CustomerId, lineItems, DateTimeOffset.UtcNow);
    }
}
```

public class WhenCreatingAnOrder
{
    [Fact]
    public void it_returns_an_order_placed_event()
    {
        var command = new CreateOrderCommand(
            CustomerId: Guid.NewGuid(),
            LineItems: [new LineItemRequest(Guid.NewGuid(), 2, 19.99m)]);

        var result = CreateOrderEndpoint.Handle(command);

        result.ShouldNotBeNull();
        result.CustomerId.ShouldBe(command.CustomerId);
        result.LineItems.Count.ShouldBe(1);
        result.LineItems[0].Quantity.ShouldBe(2);
    }

    [Fact]
    public async Task it_returns_422_when_active_order_limit_reached()
    {
        var session = Substitute.For<IDocumentSession>();
        session.Query<Order>().CountAsync(Arg.Any<Expression<Func<Order, bool>>>(), Arg.Any<CancellationToken>())
            .Returns(50);
        session.LoadAsync<Customer>(Arg.Any<Guid>(), Arg.Any<CancellationToken>())
            .Returns(new Customer());

        var command = new CreateOrderCommand(Guid.NewGuid(), [new LineItemRequest(Guid.NewGuid(), 1, 10m)]);

        var result = await CreateOrderEndpoint.LoadAsync(command, session, CancellationToken.None);

        result.ShouldNotBeNull();
        result.Status.ShouldBe(422);
    }
}
