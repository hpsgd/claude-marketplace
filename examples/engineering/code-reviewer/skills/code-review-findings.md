Review this PR. It adds a new endpoint for fetching order history with line items.

```python
# app/routes/orders.py
from fastapi import APIRouter, Query
from app.db import get_session
from app.models import Order, LineItem

router = APIRouter()

@router.get("/orders")
async def get_orders(customer_id: str = Query(...)):
    session = get_session()

    # Fetch orders for customer
    query = f"SELECT * FROM orders WHERE customer_id = '{customer_id}'"
    orders = session.execute(query).fetchall()

    results = []
    for order in orders:
        # Get line items for each order
        items = session.query(LineItem).filter(LineItem.order_id == order.id).all()

        total = sum(item.unit_price * item.quantity for item in items)
        results.append({
            "order_id": order.id,
            "status": order.status.upper(),
            "total": total,
            "items": [{"sku": i.sku, "qty": i.quantity} for i in items],
        })

    return {"orders": results}
```
